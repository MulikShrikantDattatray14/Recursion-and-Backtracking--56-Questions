function countconse(n) {
    // when n=1 , zeroend count =1 and oneend count =1;
    let zeroend = 1;// 
    let oneend = 1;//
    let sum = zeroend + oneend;// when n=1 , sum=2
    if (n == 1) return sum;
    let i = 2;
    while (i <= n) {
        oneend = zeroend % mod;
        zeroend = sum % mod;
        sum = (zeroend + oneend) % mod;
        i++;
    }
    return (sum)


}
